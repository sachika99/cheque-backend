# Use .NET 8.0 SDK for building
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy solution and project files
COPY ["MotorStores.Api/MotorStores.Api.csproj", "MotorStores.Api/"]
COPY ["MotorStores.Application/MotorStores.Application.csproj", "MotorStores.Application/"]
COPY ["MotorStores.Infrastructure/MotorStores.Infrastructure.csproj", "MotorStores.Infrastructure/"]

# Restore dependencies
RUN dotnet restore "MotorStores.Api/MotorStores.Api.csproj"

# Copy all source code
COPY . .

# Build the application
WORKDIR "/src/MotorStores.Api"
RUN dotnet build "MotorStores.Api.csproj" -c Release -o /app/build

# Publish the application
RUN dotnet publish "MotorStores.Api.csproj" -c Release -o /app/publish /p:UseAppHost=false

# Use .NET 8.0 runtime for running (smaller image)
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app
EXPOSE 8080

# Copy published app from build stage
COPY --from=build /app/publish .

# Set environment to Production
ENV ASPNETCORE_ENVIRONMENT=Production
ENV ASPNETCORE_URLS=http://+:8080

ENTRYPOINT ["dotnet", "MotorStores.Api.dll"]
